Table Collection 2 - centred on the person table


person table - enumerated fields

source_code = enum('A','E','F','G','K','M','N','P','S','T','U','Z'); e.g. A=Associates System, P=Personnel System, S=Student Information System, F=Functional Account

person_status_code = enum('F','N','P','S','U','X'); e.g. P=Postgrad, U=Undergrad, S=Staff

person_state = enum('expected','present','graduating','absent','left','missing','deleted','extended')

reg_status = enum('Y','N','E','I','R','W','F','J','S','X','C') DEFAULT NULL,

leaver_flag = enum('N','Y')

talis_hold = enum('Y','N')

title = enum ('Dr','Miss','Mr','Mrs','Ms','Prof','Rev','Sr','Lady','Sir','Fr','Eur Ing','Attorney','Visc','Major','Rt Rev','Lord')

stu_levl = enum('UG','PG','FE','PQ','CE')

stu_styp enum('C','E','N','R','T','D','F')

stu_stst enum('AS','IS','DE','IG','RR','SU','WD','WU','IN','WN') DEFAULT NULL,

stu_ests = enum('EL','SU','RS','IN','WD','WU','RR','RW','RP','IG','WN','RE','TG','RX','XP','EA','RY'); e.g. RE=Registered, RW=Repeat Without Attendance, RR=Revoked Registration

stu_att = enum('F','S','P','E','D','L','LI','O')

multi_post = enum('N','Y')
